使用JavaScript两种常规的实现方法

# no ncapsulation function

没封装函数的

var str = 'asdfssaaasasasasaa';
var json = {};
for (var i = 0; i < str.length; i++) {// 按顺遍历
  if (!json[str.charAt(i)]) {
    json[str.charAt(i)] = 1;
  } else {
    json[str.charAt(i)]++;
  }
};
var iMax = 0;
var iIndex = '';
for (var i in json) {
  if (json[i] > iMax) {
    iMax = json[i];
    iIndex = i;
  }
}
console.log('出现次数最多的是:' + iIndex + '出现' + iMax + '次');

# encapsulation

封装函数的

function num(str) {
  var a = {};
  var b = str.split("");// 切割成字符（元素）
  // key为元素   
  // value为个数
  for (let i = 0; i < b.length; i++) {
    if (a[b[i]]) {
      a[b[i]]++;
    } else {
      a[b[i]] = 1;
    }
  }
  // console.log(a)
  // 通过比较找出最大的
  var maxLetter = "";
  var levelNum = 0;
  for (const k in a) {
    if (a[k] > levelNum) {
      levelNum = a[k];
      maxLetter = k;
    }
  }
  console.log('出现次数最多的是' + maxLetter + ": " + levelNum + '次');
  return maxLetter;
}
var cc = "dafsfsfasfafaqertyyuuioll,mmnnbvvsfsdfqgsafsafgff";
num(cc);
